JavaScript // **Array to organize data** - Stores all registered users let registeredUsers = []; // **Variable to store user input or data** - An object that will hold the data of the current user let currentUserData = {}; /** * **Function that performs an action** - Gathers form data, validates it, * stores it in the array, and updates the table. * @param {Event} event - The form submission event. */ function saveUserData(event) { // Prevent the default form submission behavior (which causes a page reload) event.preventDefault(); // Gather data from the form inputs const idNumber = document.getElementById('idNumber').value; const firstName = document.getElementById('firstName').value; const middleName = document.getElementById('middleName').value; const lastName = document.getElementById('lastName').value; const gender = document.getElementById('gender').value; const birthday = document.getElementById('birthday').value; // Store the gathered data in the currentUserData **variable** (an object) currentUserData = { id: idNumber, first: firstName, middle: middleName, last: lastName, gender: gender, bday: birthday }; // Add the new user data to the **registeredUsers array** registeredUsers.push(currentUserData); // Update the webpage content dynamically displayUsersInTable(); // Clear the form after saving document.getElementById('signUpForm').reset(); } /** * **Action that updates the webpage content dynamically** - * Renders the data from the registeredUsers array into the HTML table. */ function displayUsersInTable() { const tableBody = document.getElementById('usersTableBody'); // Clear the existing content of the table body tableBody.innerHTML = ''; // Loop through the **registeredUsers array** and create a row for each user registeredUsers.forEach(user => { const row = tableBody.insertRow(); // 1. ID Number let cell = row.insertCell(); cell.textContent = user.id; // 2. Full Name (Formatted) cell = row.insertCell(); // Construct full name, only adding middle name if it exists const fullName = user.middle ? `${user.first} ${user.middle} ${user.last}` : `${user.first} ${user.last}`; cell.textContent = fullName; // 3. Gender cell = row.insertCell(); cell.textContent = user.gender; // 4. Birthday cell = row.insertCell(); cell.textContent = user.bday; }); } // --- **DOM Event Listeners** --- // 1. **DOM event (Button click)**: Attach the saveUserData function to the form's submit event. // When the form is submitted (e.g., by clicking the "Save Data" button), saveUserData runs. document.getElementById('signUpForm').addEventListener('submit', saveUserData); // 2. **DOM event (Element Click and dynamic style change)**: Add an interactive feature // to the page header, which changes its color on click. document.getElementById('page-header').addEventListener('click', function() { const currentColor = this.style.color; // Toggle the color between black and a dynamic shade of green this.style.color = (currentColor === 'rgb(40, 167, 69)' || currentColor === '') ? '#28a745' : 'black'; }); To use these files: Create a folder on your computer. Save the HTML code above into a file named formative4.html inside that folder.
